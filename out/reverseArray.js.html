/**
 * 
 * @param {Array} array input  
 * @returns {Array}reversed array 
 */
function reverseArray(array){
    let newArrays=[];
    for(let i=array.length-1;i>=0;i--){
        val= array[i];
        newArrays.push(val);   
    }
    return newArrays;

}
